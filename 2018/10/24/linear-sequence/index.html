[数据结构-教材示例]线性表（一）：顺序表

一 预定义

本书当中的预定义：

#define TRUE 1

#define FALSE 0

#define OK 1

#define ERROR 0

#define INFEASIBLE -1

#define OVERFLOW -2

typedef int Status;

顺序表中的预定义：

#define LIST_INIT_SIZE 100

#define LISTINCREMENT 10

typedef struct {
char *elem;
int length;
int listsize;
}SqList;

Status equalChar(char a, char b)
{
return a == b;
}

ADT抽象数据结构：

  • 操作结果：将 L 重置为空表。

  • 初始条件：线性表 L 已存在。
  • 操作结果：若 L 为空表，则返回 TRUE，否则返回 FALSE。

  • 初始条件：线性表 L 已存在。

  • 初始条件：线性表 L 已存在，compare()是数据元素判定函数。

  • 初始条件：线性表 L 已存在。

  • 初始条件：线性表 L 已存在，1≤i≤ListLength(L)+1。

    * ListDelete( &L, i ,&e )

  • 初始条件：线性表 L 已存在。

二 操作

Status InitList_Sq(SqList &L) {
L.elem = (char*)malloc(LIST_INIT_SIZE * sizeof(char));
if (!L.elem)
exit(OVERFLOW);
L.length = 0;
L.listsize = LIST_INIT_SIZE;
return OK;
}
//构建一个空的线性表L。

L.length表示顺序表的长度，L.listsize表示分配到的空间。

Status ListInsert_Sq(SqList&L, int i, char e)
{
if (i<1 || i>L.length + 1) return FALSE;
if (L.length >= L.listsize)//存储空间不够，重新分配。
{
char* new_base = (char)realloc(L.elem, (LIST_INIT_SIZE + LISTINCREMENT) * sizeof(char));
if (!new_base)
exit(OVERFLOW);
L.elem += LISTINCREMENT;
}
char\p, *q;
q = &(L.elem[i - 1]);
for (p = &(L.elem[L.length - 1]); p >= q; p–) *(p + 1) = *p;
//从最后一个元素开始，依次向后移位。

*q = e;
L.length++;
return OK;

}
//在L中的第i个位置之前插入新的数据元素e，L的长度加1。

先判断插入位置是否合法，再判断顺序表是否已满，若满则增加分配空间。之后现将插入位置及其后元素向后移一位，再插入元素。

Status ListDelete_Sq(SqList &L, int i, char &e)
{
if (i<1 || i>L.length) return FALSE;
char*p, *q;
p = &(L.elem[i - 1]);
e = *p;
q = L.elem + L.length - 1;//寻找表尾元素
for (++p; p <= q; ++p) *(p - 1) = *p;
–L.length;
return OK;
}
//删除L的第i个数据元素，并用e返回其值，L的长度减1。

因为要对e进行操作，所以形参处要写&。(宛如废话=-=)

int LocateElem_Sq(SqList &L, char e, Status(compare)(char, char))
{
int i = 1;
char p = L.elem;
while (i <= L.length && !(\compare)(\(p++), e)) ++i;
if (i <= L.length) return i;
else return 0;
}
//返回L中第1个与e满足关系compare()的数据元素的位序。若这样的数据元素不存在，则返回0。

这里可以自己规定compare()函数。我用的是预定义中的equalChar()，即判断两字符是否相等。

void MergeList_Sq( SqList La,SqList Lb,SqList &Lc )
{
char *pa, *pb, *pc, *pa_last, pb_last;
pa = La.elem; pb = Lb.elem;
Lc.listsize = Lc.length = La.length + Lb.length;
pc = Lc.elem = (char *)malloc(Lc.listsize * sizeof(char));
if (!Lc.elem) exit(OVERFLOW);
pa_last = La.elem + La.length - 1;
pb_last = Lb.elem + Lb.length - 1;
while (pa <= pa_last && pb <= pb_last)
{
if (\pa <= *pb) *pc++ = *pa++;
else *pc++ = *pb++;
}
while (pa <= pa_last)*pc++ = *pa++;
while (pb <= pb_last)*pc++ = *pb++;
}
//按照非递减顺序将两个非递减顺序排列的顺序表归并。

简单来说就是将两个表中的元素逐一比较，较小的放入新表，指针后移，再比较，直到一个顺序表元素全部比较完毕。之后将另一顺序表中的剩余元素直接存入新表。
